// SAMPLE C# CODE
public IEnumerator DepthFirstSearch(Node start) {
    var stack = new Stack();
    var visited = new HashSet();
    stack.Push(start);

    while (stack.Count > 0) {
        var node = stack.Pop();
        if (visited.Contains(node)) {
            continue;
        }

        Visit(node);
        visited.Add(node);

        foreach (var neighbor in node.Neighbors) {
            stack.Push(neighbor);
        }

        yield return null;
    }
}
